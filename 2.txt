#!/bin/bash

# 获取用户输入
read -p "请输入一个水果的名称： " fruit

# 使用 case 语句处理不同的水果
case $fruit in
    apple)
        echo "苹果是一种常见的水果，富含维生素C。"
        ;;
    banana)
        echo "香蕉富含钾，对心脏健康有益。"
        ;;
    orange)
        echo "橘子含有丰富的维生素C和纤维。"
        ;;
    *)
        echo "抱歉，我不认识这个水果。"
        ;;
esac
